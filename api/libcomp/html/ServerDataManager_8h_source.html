476  const libcomp::String& filePath,
477  DefinitionManager* definitionManager = nullptr)
478  {
479  tinyxml2::XMLDocument objsDoc;
480 
481  std::vector<char> data = pDataStore->ReadFile(filePath);
482 
483  if(data.empty())
484  {
485  LogServerDataManagerWarning([&]()
486  {
487  return String("File does not exist or is empty: %1\n")
488  .Arg(filePath);
489  });
490 
491  return true;
492  }
493 
494  if(tinyxml2::XML_SUCCESS !=
495  objsDoc.Parse(&data[0], data.size()))
496  {
497  return false;
498  }
499 
500  const tinyxml2::XMLElement *rootNode = objsDoc.RootElement();
501  const tinyxml2::XMLElement *objNode = rootNode->FirstChildElement("object");
502 
503  while(nullptr != objNode)
504  {
505  if(!LoadObject<T>(objsDoc, objNode, definitionManager))
506  {
507  LogServerDataManagerError([&]()
508  {
509  return String("Failed to load XML file: %1\n")
510  .Arg(filePath);
511  });
512 
513  return false;
514  }
515 
516  objNode = objNode->NextSiblingElement("object");
517  }
518 
519  LogServerDataManagerInfo([&]()
520  {
521  return String("Loaded XML file: %1\n").Arg(filePath);
522  });
523 
524  return true;
